// BILETUL 10
// Se citeste un numar natural a.
// Sa se creeze o matrice patratica de ordin a care contine:
// - cifra 4 pe diagonala principala
// - valoarea 1 pentru elementele ramase

#include <iostream>
using namespace std;
int main()
{
    int a;
    int mat[100][100];
    cout << "Introduceti ordinul matricei a = ";
    cin >> a;

    for(int i = 1; i <= a; i++)
        for(int j = 1; j <= a; j++)
        {
            if(i == j)
                mat[i][j] = 4;
            else
                mat[i][j] = 1;
        }

    cout << endl;
    cout << "Matricea generata este:" << endl;
    for(int i = 1; i <= a; i++)
    {
        for(int j = 1; j <= a; j++)
            cout << mat[i][j] << " ";
        cout << endl;
    }

    return 0;
}
