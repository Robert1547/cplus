// BILETUL 21
// Se da o matrice x cu n linii si n coloane, elemente intregi.
// Sa se determine numarul elementelor din matrice care sunt
// < 24 si se gasesc deasupra diagonalei principale, pe linii pare.

#include <iostream>
using namespace std;
int main()
{
    int n;
    int x[100][100];
    cout << "Introduceti ordinul matricei n = ";
    cin >> n;

    for(int i = 1; i <= n; i++)
        for(int j = 1; j <= n; j++)
        {
            cout << "x[" << i << "][" << j << "] = ";
            cin >> x[i][j];
        }

    cout << endl;
    cout << "Matricea introdusa este:" << endl;
    for(int i = 1; i <= n; i++)
    {
        for(int j = 1; j <= n; j++)
            cout << x[i][j] << " ";
        cout << endl;
    }

    int count = 0;
    // Deasupra diagonalei principale: j > i
    // Pe linii pare: i % 2 == 0
    for(int i = 1; i <= n; i++)
        if(i % 2 == 0)
            for(int j = i + 1; j <= n; j++)
                if(x[i][j] < 24)
                    count++;

    cout << endl;
    cout << "Numarul elementelor < 24, de deasupra diagonalei principale, pe linii pare: " << count;
    return 0;
}
