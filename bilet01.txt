// BILETUL 1
// Se citeste o matrice cu m linii si n coloane si un numar x.
// Sa se afiseze matricea si suma numerelor pare de pe linia x.

#include <iostream>
using namespace std;
int main()
{
    int m, n, x;
    int a[100][100];
    cout << "Introduceti numarul de linii m = ";
    cin >> m;
    cout << "Introduceti numarul de coloane n = ";
    cin >> n;
    cout << "Introduceti linia x = ";
    cin >> x;

    for(int i = 1; i <= m; i++)
        for(int j = 1; j <= n; j++)
        {
            cout << "a[" << i << "][" << j << "] = ";
            cin >> a[i][j];
        }

    cout << endl;
    cout << "Matricea introdusa este:" << endl;
    for(int i = 1; i <= m; i++)
    {
        for(int j = 1; j <= n; j++)
            cout << a[i][j] << " ";
        cout << endl;
    }

    int suma = 0;
    for(int j = 1; j <= n; j++)
        if(a[x][j] % 2 == 0)
            suma += a[x][j];

    cout << endl;
    cout << "Suma numerelor pare de pe linia " << x << " este: " << suma;
    return 0;
}
