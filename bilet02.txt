// BILETUL 2
// Se citeste un numar n cu max. 9 cifre.
// Sa se verifice daca toate cifrele numarului n sunt pare.

#include <iostream>
using namespace std;
int main()
{
    long long n;
    cout << "Introduceti numarul n = ";
    cin >> n;

    long long temp = (n < 0) ? -n : n;
    bool toatePare = true;

    while(temp > 0)
    {
        int cifra = temp % 10;
        if(cifra % 2 != 0)
        {
            toatePare = false;
            break;
        }
        temp /= 10;
    }

    cout << endl;
    if(toatePare)
        cout << "Toate cifrele numarului sunt pare.";
    else
        cout << "Nu toate cifrele numarului sunt pare.";

    return 0;
}
