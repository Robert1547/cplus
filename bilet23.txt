// BILETUL 23
// Se da o matrice x cu n linii si n coloane.
// Sa se determine maximul elementelor pare de pe coloana p.

#include <iostream>
#include <climits>
using namespace std;
int main()
{
    int n, p;
    int x[100][100];
    cout << "Introduceti ordinul matricei n = ";
    cin >> n;

    for(int i = 1; i <= n; i++)
        for(int j = 1; j <= n; j++)
        {
            cout << "x[" << i << "][" << j << "] = ";
            cin >> x[i][j];
        }

    cout << "Introduceti coloana p = ";
    cin >> p;

    int maxPar = INT_MIN;
    bool gasit = false;

    for(int i = 1; i <= n; i++)
        if(x[i][p] % 2 == 0)
        {
            if(!gasit || x[i][p] > maxPar)
            {
                maxPar = x[i][p];
                gasit = true;
            }
        }

    cout << endl;
    if(gasit)
        cout << "Maximul elementelor pare de pe coloana " << p << " este: " << maxPar;
    else
        cout << "Nu exista elemente pare pe coloana " << p << ".";

    return 0;
}
